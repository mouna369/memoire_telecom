spark-worker-3:
    image: apache/spark:latest
    container_name: worker_3_pfe
    environment:
      - SPARK_MODE=worker
      - SPARK_MASTER_URL=spark://spark-master:7077
      - PYTHONPATH=/opt/pymongo_libs
    volumes:
      - spark-data:/data
      - ./data/results:/opt/data
      - ./pymongo_libs:/opt/pymongo_libs
    networks:
      - spark_network
    depends_on:
      - spark-master
    command: /opt/spark/bin/spark-class org.apache.spark.deploy.worker.Worker spark://spark-master:7077
    restart: always











# Redémarrer avec le nouveau Worker
docker compose down
docker compose up -d
